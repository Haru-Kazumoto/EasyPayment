CREATE TABLE users (
id INT AUTO_INCREMENT PRIMARY KEY,
email VARCHAR(100),
password VARCHAR(255),
student_id INT
);


CREATE TABLE students (
id INT AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(100),
class VARCHAR(50)
);


CREATE TABLE bills (
id INT AUTO_INCREMENT PRIMARY KEY,
student_id INT,
name VARCHAR(100),
amount INT,
status ENUM('unpaid','paid')
);
